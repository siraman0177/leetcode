class Solution {
public:
    int majorityElement(vector<int>& nums) {
       int n=nums.size();

        for (int val:nums){
            int flag =0;
            for(int vall:nums){
                if(vall==val){
                flag++;}
            }
        if(flag>n/2){
        return val ;}
        }
    return -1;
    }
};

# In brute froce approach we just counted the frequency and return the value
