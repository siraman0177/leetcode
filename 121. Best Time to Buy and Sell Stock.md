class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        if (n == 0) return 0;

        int mini = prices[0];
        int maxi = 0;

        for(int i = 1; i < n; i++) {
            if (prices[i] < mini)
                mini = prices[i];
            else {
                int profit = prices[i] - mini;
                if (profit > maxi)
                    maxi = profit;
            }
        }
        return maxi;
    }
};
